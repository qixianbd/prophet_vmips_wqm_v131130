00093 {
00094         uint32 word = 0;
00095         
00096         switch( offset / 4 ) {
00097         case 0:         // keyboard 1 control
00098                 word = keyboard_interrupt_enable[0] ? CTL_IE : 0;
00099                 if( line_connected(0) )
00100                         word |= lines[0].keyboard_state;
00101                 if (!keyboard_interrupt_enable[0])
00102                         deassertInt (IRQ3);     
00103                 break;
00104         case 1:         // keyboard 1 data
00105                 if( line_connected(0) ) {
00106                         unready_keyboard( 0 );
00107                         word = lines[0].keyboard_char;
00108                 }
00109                 break;
00110         case 2:         // display 1 control
00111                 word = display_interrupt_enable[0] ? CTL_IE : 0;
00112                 if( line_connected(0) )
00113                         word |= lines[0].display_state;
00114                 else
00115                         word |= CTL_RDY;
00116                 if (!display_interrupt_enable[0])
00117                         deassertInt (IRQ4);     
00118                 break;
00119         case 3:         // display 1 data
00120                 break;
00121         case 4:         // keyboard 2 control
00122                 word = keyboard_interrupt_enable[1] ? CTL_IE : 0;
00123                 if( line_connected(1) )
00124                         word |= lines[1].keyboard_state;
00125                 if (!keyboard_interrupt_enable[1])
00126                         deassertInt (IRQ5);     
00127                 break;
00128         case 5:         // keyboard 2 data
00129                 if( line_connected(1) ) {
00130                         unready_keyboard( 1 );
00131                         word = lines[1].keyboard_char;
00132                 }
00133                 break;
00134         case 6:         // display 2 control
00135                 word = display_interrupt_enable[1] ? CTL_IE : 0;
00136                 if( line_connected(1) )
00137                         word |= lines[1].display_state;
00138                 else
00139                         word |= CTL_RDY;
00140                 if (!display_interrupt_enable[1])
00141                         deassertInt (IRQ6);     
00142                 break;
00143         case 7:         // display 2 data
00144                 break;
00145         case 8:         // clock control
00146                 word = clock_interrupt ? CTL_IE : 0;

00158 {
00159         data = machine->physmem->host_to_mips_word(data);

00161         switch( offset / 4 ) {
00162         case 0:         // keyboard 1 control
00163                 keyboard_interrupt_enable[0] = data & CTL_IE;
00164                 if( line_connected(0) && keyboard_interrupt_enable[0]
00165                     && lines[0].display_state == READY )
00166                         assertInt( IRQ3 );
00167                 break;
00168         case 1:         // keyboard 1 data
00169                 break;
00170         case 2:         // display 1 control
00171                 display_interrupt_enable[0] = data & CTL_IE;
00172                 if( line_connected(0) && display_interrupt_enable[0]
00173                     && lines[0].display_state == READY )
00174                         assertInt( IRQ4 );
00175                 break;
00176         case 3:         // display 1 data
00177                 if( line_connected(0) )
00178                         unready_display( 0, data );
00179                 break;
00180         case 4:         // keyboard 2 control
00181                 keyboard_interrupt_enable[1] = data & CTL_IE;
00182                 if( line_connected(1) && keyboard_interrupt_enable[1] &&
00183                     lines[1].keyboard_state == READY )
00184                         assertInt( IRQ5 );
00185                 break;
00186         case 5:         // keyboard 2 data
00187                 break;
00188         case 6:         // display 2 control
00189                 display_interrupt_enable[1] = data & CTL_IE;
00190                 if( line_connected(1) && display_interrupt_enable[1] &&
00191                     lines[1].display_state == READY )
00192                         assertInt( IRQ6 );
00193                 break;
00194         case 7:         // display 2 data
00195                 if( line_connected(1) )
00196                         unready_display( 1, data );
00197                 break;
00198         case 8:         // clock control
00199                 clock_interrupt = data & CTL_IE;
00200                 if( clock_interrupt && clock_state == READY )
00201                         assertInt( IRQ2 );
00202                 break;
00203         default:
