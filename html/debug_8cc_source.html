00295 Debug::targetloop(void)
00296 {
00297         int packetno = 0;
00298         char buf[PBUFSIZ];
00299         char *result = NULL;
00300 
00301         while (! machine->halted) {
00302                 exception_pending = false;
00303                 /* Wait for a packet, and when we get it, store it in
00304                  * BUF. If we get an error trying to receive a packet,
00305                  * give up.
00306                  */
00307                 getpkt(buf, 1); if (remotegdb_backend_error) return;
00308                 switch(buf[0]) {
00309                         case 'g': result = target_read_registers(buf); break;
00310                         case 'G': result = target_write_registers(buf); break;
00311                         case 'm': result = target_read_memory(buf); break;
00312                         case 'M': result = target_write_memory(buf); break;
00313                         case 'c': result = target_continue(buf); break;
00314                         case 's': result = target_step(buf); break;
00315                         case 'k': result = target_kill(buf); break;
00316                         case 'H': result = target_set_thread(buf); break;
00317                         case '?': result = target_last_signal(buf); break;
00318                         case 'Z': result = target_set_or_remove_breakpoint(buf, true);
00319                       break;
00320                         case 'z': result = target_set_or_remove_breakpoint(buf, false);
00321                       break;
00322                         default:  result = target_unimplemented(buf); break;
00323                 }
00324                 packetno++;
00325                 putpkt(result); if (remotegdb_backend_error) return;
00326                 free(result);
00327         }

00603 /* Translates between MIPS exception codes and Unix signals (which GDB
00604  * likes better.)

00606 int
00607 Debug::exccode_to_signal(int exccode)
00608 {
00609         const int signos[] = { 2,  /* Int -> SIGINT */
00610          11, 11, 11, 11, 11,       /* Mod, TLBL, TLBS, AdEL, AdES -> SIGSEGV */
00611          7, 7,                     /* IBE, DBE -> SIGBUS */
00612          5, 5,                     /* Sys, Bp -> SIGTRAP */
00613          4,                        /* RI -> SIGILL */
00614          8, 8 };                   /* CpU, OV -> SIGFPE */
00615         const int defaultsig = 1;  /* others -> SIGHUP */
00616   
00617         if (exccode < Int || exccode > Ov) {
00618                 return defaultsig; /* SIGHUP */
00619         } else {
00620                 return signos[exccode];
00621         }
