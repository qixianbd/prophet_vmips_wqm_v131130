00082         {

00091                 while (1 < len) {               /* Reserve space for null */
00092                         XtAppNextEvent (appcontext, &event);
00093                         if (event.type == KeyPress) {
00094                                 int chars = XLookupString (&event.xkey, buffer, 10, &key, &compose);
00095                                 if ((key == XK_Return) || (key == XK_KP_Enter)) {
00096                                         *ptr++ = '\n';

00098                                         output("\n", 0, 1);
00099                                         break;
00100                                 }
00101                                 else if (*buffer == 3) /* ^C */
00102                                         XtDispatchEvent (&event);
00103                                 else if (chars > 0) {
00104                                         /* Event is a character, not a modifier key */
00105                                         int n = (chars < len - 1 ? chars : len - 1);
00106 
00107                                         strncpy (ptr, buffer, n);
00108                                         ptr += n;
00109                                         len -= n;
00110 
00111                                         buffer[chars] = '\0';
00112                                         output(buffer, 0, chars);
00113                                 }
00114                         }else
00115                                 XtDispatchEvent (&event);
00116                 }
00117 
00118                 if (0 < len)
00119                         *ptr = '\0';
