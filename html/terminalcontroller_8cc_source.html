00068 bool TerminalController::connect_terminal( int tty_fd, int line ) throw()
00069 {
00070         // there is already a terminal connected to that line
00071         if( line_connected( line ) ) {
00072                 while( close( tty_fd ) == -1 && errno == EINTR );
00073                 return false;
00074         }
00075         assert( !lines[line].keyboard_repoll );
00076         assert( !lines[line].display_delay );
00077 
00078         if( !isatty( tty_fd ) ) {
00079                 error( "attempt to connect non tty to line %d", line );
00080                 while( close( tty_fd ) == -1 && errno == EINTR );
00081                 return false;
00082         }
00083 
00084         // save the old terminal state
00085         if( tcgetattr( tty_fd, &lines[line].tty_state ) == -1 ) {
00086                 error( "cannot get terminal state on line %d: %s",
00087                        line, strerror(errno) );
00088                 while( close( tty_fd ) == -1 && errno == EINTR );
00089                 return false;
00090         }
00091 
00092         // prepare to call prepare_tty()
00093         lines[line].tty_fd = tty_fd;
00094 
00095         if( !prepare_tty( line ) ) {
00096                 lines[line].tty_fd = -1;
00097                 while( close( tty_fd ) == -1 && errno == EINTR );
00098                 return false;
00099         }
00100 
00101         lines[line].keyboard_char = 0;
00102         lines[line].keyboard_state = UNREADY;
00103         lines[line].display_state = READY;
00104 
00105         // the keyboard starts with no data
00106         FD_SET( tty_fd, &unready_keyboards );
00107 
00108         if( tty_fd + 1 > max_fd )
00109                 max_fd = tty_fd + 1;
00110 
00111         assert( line_connected( line ) );
00112         return true;
00113 }

00115 void TerminalController::remove_terminal( int line ) throw()
00116 {
00117         assert( line >= 0 && line < MAX_TERMINALS );
00118 
00119         int tty_fd = lines[line].tty_fd;
00120         if( tty_fd == -1 ) {
00121                 assert( !lines[line].keyboard_repoll );
00122                 assert( !lines[line].display_delay );
00123                 return;
00124         }
00125         assert( isatty( lines[line].tty_fd ) );
00126 
00127         // invalidate this line
00128         lines[line].tty_fd = -1;
00129         FD_CLR( tty_fd, &unready_keyboards );
00130 
00131         // reset the original terminal settings
00132         if( tcsetattr( tty_fd, TCSAFLUSH, &lines[line].tty_state ) == -1 )
00133                 error( "cannot restore state for terminal on line %d", line );
00134         
00135         // make sure to cancel tasks which refer to this line
00136         if( lines[line].keyboard_repoll ) {
00137                 lines[line].keyboard_repoll->cancel();
00138                 lines[line].keyboard_repoll = NULL;
00139         }
00140 
00141         if( lines[line].display_delay ) {
00142                 lines[line].display_delay->cancel();
00143                 lines[line].display_delay = NULL;
00144         }
00145 
00146         // recompute max_fd
00147         max_fd = -1;
00148         for( int i = 0; i < MAX_TERMINALS; i++ ) {
00149                 if( lines[i].tty_fd > max_fd )
00150                         max_fd = lines[i].tty_fd;
00151         }
00152         if( max_fd != -1 )
00153                 max_fd++;
00154         
00155         assert( !line_connected( line ) );
