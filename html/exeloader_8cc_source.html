00084 static bool file_is_elf (FILE *fp) {
00085   fseek (fp, 0, SEEK_SET);
00086   char buf[4];
00087   if (fread (&buf, sizeof(char), 4, fp) != 4)
00088     return false;
00089   return (buf[0] == 0x7f) && (buf[1] == 'E') && (buf[2] == 'L')
00090          && (buf[3] == 'F');

00099 char *vmips::translate_to_host_ram_pointer (uint32 vaddr) {
00100   // Translate vaddr to physical address.
00101   uint32 paddr;
00102   if ((vaddr & KSEG_SELECT_MASK) == KSEG0) {
00103     paddr = vaddr - KSEG0_CONST_TRANSLATION;
00104   } else if ((vaddr & KSEG_SELECT_MASK) == KSEG1) {
00105     paddr = vaddr - KSEG1_CONST_TRANSLATION;
00106   } else {
00107     error ("Virtual address 0x%x is not in KSEG0 or KSEG1", vaddr);
00108     return 0;
00109   }
00110 
00111   // Get pointer to where physaddr is in RAM and return it.
00112   if (memmod->incorporates(paddr)) {
00113     return (char *) memmod->getAddress() + (paddr - memmod->getBase());
00114   } else {
00115     error ("Virtual address 0x%x is not within physical RAM", vaddr);
00116     return 0;
00117   }
00118 }

00120 bool vmips::load_ecoff (FILE *fp) {
00121   coff_section_header *sections = 0;
00122   bool rv = true;
00123   try {
00124 
00125   if (fseek (fp, 0, SEEK_SET) < 0)
00126     throw std::string ("Can't seek to file header");
00127   coff_info header;
00128   if (fread (&header.file_header, sizeof (coff_file_header), 1, fp) != 1)
00129     throw std::string ("Can't read file header");
00130 
00131   if (fseek (fp, 76, SEEK_SET) < 0)
00132     throw std::string ("Can't seek to section headers");
00133   int nsects = header.file_header.no_sections;
00134   sections = new coff_section_header[nsects];
00135   for (int i = 0; i < nsects; ++i)
00136     if (fread (&sections[i], sizeof (coff_section_header), 1, fp) != 1)
00137       throw std::string ("Can't read section headers");
00138 
00139   coff_section_header *section = sections;
00140   for (int i = 0; i < header.file_header.no_sections; i++) {
00141     if (strlen(section->section_name) != 0)  {
00142       if (strcmp(section->section_name, ".text") == 0) {
00143         header.text_addr = section->section_virt_addr;
00144         header.text_offset = section->section_file_loc;
00145         header.text_size = section->section_size;
00146       } else if (strcmp(section->section_name, ".data") == 0) {
00147         header.data_addr = section->section_virt_addr;
00148         header.data_offset = section->section_file_loc;
00149         header.data_size = section->section_size;
00150       } else if (strcmp(section->section_name, ".bss") == 0) {
00151         header.bss_addr = section->section_virt_addr;
00152         header.bss_size = section->section_size;
00153       }
00154     }
00155     section++;
00156   }
00157   printf
00158     ("size: text %lx data %lx bss %lx entry %lx\n"
00159      "base: text %lx data %lx bss %lx\n" "gp: %lx\n",
00160      header.file_header.text_size, header.file_header.data_size,
00161      header.file_header.bss_size, header.file_header.entry_addr,
00162      header.file_header.text_start, header.file_header.data_start,
00163      header.file_header.bss_start, header.file_header.gp_value);
00164   printf ("sections: .text@0x%x, %u bytes, at file offset %u\n",
00165           (unsigned int) header.text_addr, (unsigned int) header.text_size,
00166           (unsigned int) header.text_offset);
00167   printf ("          .data@0x%x, %u bytes, at file offset %u\n",
00168           (unsigned int) header.data_addr, (unsigned int) header.data_size,
00169           (unsigned int) header.data_offset);
00170   printf ("          .bss@0x%x, %u bytes\n", (unsigned int) header.bss_addr,
00171           (unsigned int) header.bss_size);
00172   putchar ('\n');
00173 
00174   char *text_area_dst = translate_to_host_ram_pointer (header.text_addr);
00175   char *data_area_dst = translate_to_host_ram_pointer (header.data_addr);
00176   char *bss_area_dst = translate_to_host_ram_pointer (header.bss_addr);
00177   printf ("host ptrs; text=%p data=%p bss=%p\n", text_area_dst, data_area_dst,
00178           bss_area_dst);
00179   
00180   if (fseek (fp, header.text_offset, SEEK_SET) < 0)
00181     throw std::string ("Can't seek to text offset");
00182   if (fread (text_area_dst, 1, header.text_size, fp) != header.text_size)
00183     throw std::string ("Can't read text");
00184   if (fseek (fp, header.data_offset, SEEK_SET) < 0)
00185     throw std::string ("Can't seek to data offset");
00186   if (fread (data_area_dst, 1, header.data_size, fp) != header.data_size)
00187     throw std::string ("Can't read data");
00188   memset (bss_area_dst, 0, header.bss_size);
00189   //cpu->debug_set_pc (header.file_header.entry_addr);
00190   
00191   } catch (std::string &errorstr) {
00192     error ("%s", errorstr.c_str ());
00193     rv = false;
00194   }
00195 
00196   fclose (fp);
00197   if (sections) delete sections;
00198   return rv;

00208 vmips::setup_exe ()
00209 {
00210   // Open executable file.
00211   if (strcmp (opt_execname, "none") == 0)
00212     return true;
00213   FILE *fp = fopen (opt_execname, "rb");
00214   if (!fp) {
00215     error ("Could not open executable `%s': %s", opt_execname,
00216            strerror (errno));
00217     return false;
00218   }
00219   if (file_is_ecoff (fp)) {
00220     return load_ecoff (fp);
00221   } else if (file_is_elf (fp)) {
00222     return load_elf (fp);
00223   } else {
00224     error ("Could not determine type of executable `%s'", opt_execname);
00225     return false;
00226   }
