00038 int
00039 Options::tilde_expand(char *filename)
00040 {
00041         struct passwd *pw;
00042         char buf[PATH_MAX], *trailer = NULL;
00043 
00044         if (filename[0] != '~') {
00045                 return 0;
00046         }
00047         filename[0] = '\0';
00048         trailer = strchr(&filename[1],'/');
00049         if (trailer) {
00050                 *trailer++ = '\0';
00051         }
00052         /* &filename[1] now consists of mumble\0 where mumble = zero or more chars
00053          * representing a username
00054          */
00055         if (strlen(&filename[1]) == 0) {
00056                 pw = getpwuid(getuid());
00057         } else {
00058                 pw = getpwnam(&filename[1]);
00059         }
00060         if (!pw) {
00061                 return -1;
00062         }
00063         if (trailer) {
00064                 sprintf(buf,"%s/%s",pw->pw_dir,trailer);
00065         } else {
00066                 strcpy(buf,pw->pw_dir);
00067         }
00068         strcpy(filename, buf);
00069         return 0;
00070 }

00158 void
00159 Options::process_one_option(const char *const option)
00160 {
00161         char *copy = strdup(option), *equals = NULL, *trailer = NULL;
00162         uint32 num;
00163 
00164         if ((equals = strchr(copy, '=')) == NULL) {
00165                 /* FLAG option */
00166                 char *name;
00167                 bool value;
00168                 if ((trailer = strprefix(copy, "no")) == NULL) {
00169                         /* FLAG set to TRUE */
00170                         name = copy;
00171                         value = true;
00172                 } else {
00173                         /* FLAG set to FALSE */
00174                         name = trailer;
00175                         value = false;
00176                 }
00177                 if (find_option_type(name) == FLAG)
00178                                 set_flag_option(name, value);
00179                         else
00180                                 error("Unknown option: %s\n", name);
00181         } else {
00182                 /* STR or NUM option */
00183                 *equals++ = '\0';
00184                 if (*equals == '\0') {
00185                     error ("Option value missing for %s", copy);
00186                     free(copy);
00187                     return;
00188                 }
00189                 switch(find_option_type(copy)) {
00190                         case STR:
00191                                 set_str_option(copy, equals);
00192                                 break;
00193                         case NUM:
00194                                 {
00195                                         char *endptr;
00196                                         num = strtoul(equals, &endptr, 0);
00197                                         switch (*endptr) {
00198                                                 case 'k':
00199                                                 case 'K':
00200                                                         num *= 1024;
00201                                                         break;
00202                                                 case 'm':
00203                                                 case 'M':
00204                                                         num *= 1024 * 1024;
00205                                                         break;
00206                                                 case 'g':
00207                                                 case 'G':
00208                                                         num *= 1024 * 1024 * 1024;
00209                                                         break;
00210                                                 case '\0':
00211                                                         break;
00212                                                 default:
00213                                                         error("bogus suffix on numeric option\n");
00214                                                         break;
00215                                         }
00216                                         set_num_option(copy, num);
00217                                 }
00218                                 break;
00219                         default:
00220                                 *--equals = '=';
00221                                 error("Unknown option: %s\n", copy);
00222                                 break;
00223                 }
00224         }
00225         free(copy);
00226 }

00231 int
00232 Options::process_first_option(char **bufptr, int lineno, char *filename)
00233 {
00234         char *out, *in, copybuf[OPTBUFSIZ], char_seen;
00235         bool quoting, quotenext, done, string_done;
00236         
00237         out = *bufptr;
00238         in = copybuf;
00239         done = string_done = quoting = quotenext = false;
00240 
00241         while (isspace(*out)) {
00242                 out++;
00243         }
00244         while (!done) {
00245                 char_seen = *out++;
00246                 if (char_seen == '\0') {
00247                         done = true;
00248                         string_done = true;
00249                 } else {
00250                         if (quoting) {
00251                                 if (char_seen == '\'') {
00252                                         quoting = false;
00253                                 } else {
00254                                         *in++ = char_seen;
00255                                 }
00256                         } else if (quotenext) {
00257                                 *in++ = char_seen;
00258                                 quotenext = false;
00259                         } else {
00260                                 if (char_seen == '\'') {
00261                                         quoting = true;
00262                                 } else if (char_seen == '\\') {
00263                                         quotenext = true;
00264                                 } else if (char_seen == '#') {
00265                                         done = true;
00266                                         string_done = true;
00267                                 } else if (!isspace(char_seen)) {
00268                                         *in++ = char_seen;
00269                                 } else {
00270                                         done = true;
00271                                         string_done = (out[0] == '\0');
00272                                 }
00273                         }
00274                 }
00275         }
00276         *in++ = '\0';
00277         *bufptr = out;
00278         if (quoting) {
00279                 fprintf(stderr,
00280                         "warning: unterminated quote in config file %s, line %d\n",
00281                         filename, lineno);
00282         }
00283         if (strlen(copybuf) > 0) {
00284                 process_one_option(copybuf);
00285         }
00286         return string_done ? 0 : 1;
00287 }

00289 int
00290 Options::process_options_from_file(char *filename)
00291 {
00292         char *buf = new char[OPTBUFSIZ];
00293         if (!buf) {
00294                 fatal_error("Can't allocate %u bytes for I/O buffer; aborting!\n",
00295                             OPTBUFSIZ);
00296         }
00297         FILE *f = fopen(filename, "r");
00298         if (!f) {
00299                 if (errno != ENOENT) {
00300                     error ("Can't open config file '%s': %s\n", filename,
00301                            strerror(errno));
00302                 }
00303                 return -1;
00304         }
00305         char *p = buf;
00306         int rc, lineno = 1;
00307         for (char *ptr = fgets(buf, OPTBUFSIZ, f); ptr;
00308                 ++lineno, p = buf, ptr = fgets(buf, OPTBUFSIZ, f))
00309                         do
00310                                 rc = process_first_option(&p, lineno, filename);
00311                         while (rc == 1);
00312         fclose(f);
00313         delete [] buf;
00314         return 0;
00315 }

00338 void
00339 Options::process_options(int argc, char **argv)
00340 {
00341         /* Get options from defaults. */
00342         process_defaults();
00343 
00344         /* Get default name of user's config file */
00345         char user_config_filename[PATH_MAX] = "~/.vmipsrc";
00346         
00347         /* Process command line */
00348         bool read_system_config_file = true;
00349         std::vector<std::string> command_line_options;
00350         for (int i = 1; i < argc; ++i) {
00351                 if (strcmp (argv[i], "--version") == 0) {
00352                         print_package_version (PACKAGE, VERSION);
00353                         exit (0);
00354                 } else if (strcmp (argv[i], "--help") == 0) {
00355                         usage (argv[0]);
00356                         exit (0);
00357                 } else if (strcmp (argv[i], "--print-config") == 0) {
00358                         print_config_info ();
00359                         exit (0);
00360                 } else if (strcmp (argv[i], "-o") == 0) {
00361                         if (argc <= i + 1)
00362                                 error_exit ("The -o flag requires an argument. Try %s --help",
00363                             argv[0]);
00364                         command_line_options.push_back (argv[i + 1]);
00365                         ++i;
00366                 } else if (strcmp (argv[i], "-F") == 0) {
00367                         if (argc <= i + 1)
00368                                 error_exit ("The -F flag requires an argument. Try %s --help",
00369                             argv[0]);
00370                         strcpy (user_config_filename, argv[i + 1]);
00371                         ++i;
00372                 } else if (strcmp (argv[i], "-n") == 0) {
00373                         read_system_config_file = false;
00374                 } else if (i == argc - 1) {
00375                         if (!can_read_file (argv[i]) && argv[i][0] == '-') {
00376                                         error_exit ("Unrecognized option %s. Try %s --help", argv[i],
00377                                         argv[0]);
00378                         }
00379                         set_str_option("romfile", argv[i]);
00380                 } else {
00381                         error_exit ("Unrecognized option %s. Try %s --help", argv[i],
00382                         argv[0]);
00383                 }
00384         }
00385         
00386         /* Get options from system configuration file */
00387         if (read_system_config_file)
00388                 process_options_from_file (SYSTEM_CONFIG_FILE);
00389         
00390         /* Get options from user configuration file */
00391         tilde_expand (user_config_filename);
00392         process_options_from_file (user_config_filename);
00393         
00394         /* Process -o options saved from command line, above. */
00395         for (std::vector<std::string>::iterator i = command_line_options.begin (),
00396                  e = command_line_options.end (); i != e; ++i)
00397                 process_one_option (i->c_str ());
00398 }
