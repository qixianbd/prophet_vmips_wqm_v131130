00043 DECRTCDevice::DECRTCDevice (Clock *_clock, uint32 _irq) :
00044   clock(_clock), frequency_ns (0), irq (_irq), interrupt_enable (false)
00045 {
00046   extent = 0x80000;
00047   // Initialize registers
00048   for (int i = 0; i < 64; ++i) {
00049     rtc_reg[i] = 0x00;
00050   }
00051   rtc_reg[RTC_REGA] = 0x20;  // Timebase divisor = 010 (32.768 kHz)
00052   // Set up write masks
00053   for (int i = 0; i < 64; ++i) {
00054     write_masks[i] = 0xff;
00055   }
00056   write_masks[RTC_REGA] = 0x7f;
00057   write_masks[RTC_REGB] = 0xd7;
00058   write_masks[RTC_REGC] = 0x00;
00059   write_masks[RTC_REGD] = 0x00;
00060 }

00068 uint32
00069 DECRTCDevice::fetch_word(uint32 offset, int mode, DeviceExc *client)
00070 {
00071   uint32 reg_no = (offset / 4);
00072   uint32 rv = 0;
00073   if (reg_no == RTC_REGA) {
00074     // Fake "update in progress" once a second
00075     time_t real_nanos = clock->get_time().tv_nsec;
00076     if ((real_nanos + ((244 + 1948)*1000)) > 1000000000) {
00077           rtc_reg[RTC_REGA] |= REGA_UIP;
00078         } else {
00079           rtc_reg[RTC_REGA] &= ~REGA_UIP;
00080         }
00081   } 
00082   if (reg_no < 10)
00083     update_host_time ();
00084   if (reg_no < 64)
00085     rv = rtc_reg[reg_no];
00086   if (reg_no == RTC_REGC)
00087     unready_clock ();
00088   return rv;
00089 }

00091 void
00092 DECRTCDevice::update_host_time ()
00093 {
00094   const struct tm *host_time = get_host_time ();
00095   rtc_reg[RTC_SEC]  = host_time->tm_sec; /* second 0..59 */
00096   rtc_reg[RTC_MIN]  = host_time->tm_min; /* minute 0..59 */
00097   rtc_reg[RTC_HOUR] = host_time->tm_hour; /* hour 0..23 */
00098   rtc_reg[RTC_DOW]  = 1 + host_time->tm_wday; /* day of week 1..7 */
00099   rtc_reg[RTC_MON]  = 1 + host_time->tm_mon; /* month 1..12 */
00100   rtc_reg[RTC_YEAR] = host_time->tm_year % 100; /* year 0..99 */
00101 }

00103 void
00104 DECRTCDevice::store_word(uint32 offset, uint32 data, DeviceExc *client)
00105 {
00106   bool old_interrupt_enable = rtc_reg[RTC_REGB] & REGB_PIE;
00107   uint32 reg_no = (offset / 4);
00108   fprintf (stderr, "RTC %s (%u) written with 0x%x\n",
00109         reg_names[(reg_no > 14) ? 14 : reg_no], reg_no, data);
00110   if (reg_no < 64)
00111     rtc_reg[reg_no] = rtc_reg[reg_no] | (((uint8) data) & write_masks[reg_no]);
00112   if (reg_no == RTC_REGA) {
00113     /* Maybe they changed the interrupt rate selector. */
00114     uint8 rate_selector = rtc_reg[RTC_REGA] & REGA_RSX;
00115     fprintf (stderr, "RTC rate_selector set to 0x%x (%u)\n", rate_selector,
00116           int_freqs[rate_selector]);
00117     frequency_ns = int_freqs[rate_selector];
00118   }
00119   if (reg_no == RTC_REGB) {
00120     /* Maybe they enabled or disabled interrupts. */
00121     bool new_interrupt_enable = rtc_reg[RTC_REGB] & REGB_PIE;
00122     fprintf (stderr, "RTC interrupt_enable set to %d\n", new_interrupt_enable);
00123     if ((!old_interrupt_enable) && new_interrupt_enable) {
00124       fprintf (stderr, "RTC turning interrupts on\n");
00125       ready_clock ();
00126     } else if (old_interrupt_enable && (!new_interrupt_enable)) {
00127       fprintf (stderr, "RTC turning interrupts off\n");
00128     }
00129     interrupt_enable = new_interrupt_enable;
00130   }
00131 }
