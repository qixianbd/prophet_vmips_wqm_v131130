00197         {
00198                 std::string message = os->str();

00205                 case PLOG_LEVEL_WARNING:        prefix << "WARNING: "; break;
00206                 case PLOG_LEVEL_ERROR:          prefix << "ERROR: "; break;
00207                 default:                        prefix << "XXX: "; break;
00208                 }
00209                 if(Setting::Get()->m_PrintLocation)
00210                         prefix << site.m_File << "(" << site.m_Line << ") : ";
00211                 if(site.m_ClassInfo != typeid(NoClassInfo))
00212                         prefix << ClassName(site.m_ClassInfo) << "::";
00213                 prefix << site.m_Function << ": " << message;
00214                 message = prefix.str();
00215                 WriteLog(site.m_Level, message);
00216 
00217                 if(site.m_Level == PLOG_LEVEL_ERROR && Setting::Get()->m_CrashFunction)
00218                         Setting::Get()->m_CrashFunction(message);
