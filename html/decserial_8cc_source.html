00089 uint32
00090 DECSerialDevice::fetch_word (uint32 offset, int mode, DeviceExc *client)
00091 {
00092   uint32 rv = 0;
00093   switch (offset & 0x18) {
00094   case DZ_CSR:
00095     csr &= ~(DZ_CSR_RDONE | DZ_CSR_TRDY | DZ_CSR_TLINE);
00096     if (csr & DZ_CSR_MSE)
00097       // Scan for lines with data.
00098       for (int line = 0; line < 4; ++line) {
00099         if (receiver_done (line))
00100           csr |= DZ_CSR_RDONE;
00101         if (transmitter_ready (line)) {
00102           csr |= DZ_CSR_TRDY;
00103           csr |= CSR_TLINE(line);
00104         }
00105       }
00106         rv = csr;
00107     break;
00108   case DZ_RBUF:
00109     rbuf &= ~DZ_RBUF_DVAL;
00110     for (int line = 0; line < 4; ++line) {
00111       if (receiver_done (line)) {
00112         unready_keyboard (line);
00113         rbuf = lines[line].keyboard_char | DZ_RBUF_DVAL;
00114         rbuf |= RBUF_RLINE(line);
00115         break;
00116       }
00117     }
00118     rv = rbuf;
00119     break;
00120   case DZ_TCR:
00121     rv = tcr;
00122     /* fprintf (stderr, "DZ11 TCR read as 0x%x\n", rv); */
00123     break;
00124   case DZ_MSR:
00125     rv = msr;
00126     fprintf (stderr, "DZ11 MSR read as 0x%x\n", rv);
00127     break;
00128   }
00129   return machine->physmem->mips_to_host_word(rv);
00130 }

00142 void
00143 DECSerialDevice::store_word (uint32 offset, uint32 data, DeviceExc *client)
00144 {
00145   data = machine->physmem->host_to_mips_word(data);
00146   uint16 data16 = data & 0x0ffff;
00147   switch (offset & 0x18) {
00148     case DZ_CSR:
00149       fprintf (stderr, "DZ11 write CSR as %x\n", data16);
00150       csr = data16;
00151       if (csr & DZ_CSR_CLR)
00152         master_clear ();
00153       display_interrupt_enable = (csr & DZ_CSR_TIE);
00154       keyboard_interrupt_enable = (csr & DZ_CSR_RIE);
00155       fprintf (stderr, "DZ11 Keyboard IE is now %s, Display IE now %s, "
00156                "selected tx line now %d\n",
00157                keyboard_interrupt_enable ? "on" : "off",
00158                display_interrupt_enable ? "on" : "off",
00159                GET_CURRENT_CSR_TLINE (csr));
00160       break;
00161     case DZ_LPR:
00162       fprintf (stderr, "DZ11 write LPR as %x\n", data16);
00163       lpr = data16;
00164       break;
00165     case DZ_TCR:
00166       /* fprintf (stderr, "DZ11 write TCR as %x\n", data16); */
00167       tcr = data16;
00168       break;
00169     case DZ_TDR: {
00170       int line = 3; // FIXME: should be GET_CURRENT_CSR_TLINE (csr);
00171       if (line_connected (line))
00172         unready_display (line, data16 & 0xff);
00173       break;
00174     }
00175   }
00176 
00177   // Check whether we have to assert or deassert the CSR IRQ now because
00178   // flags changed.
00179   bool any_enable_is_on = (csr & (DZ_CSR_RIE | DZ_CSR_TIE));
00180   if (!any_enable_is_on) {
00181     // They turned all the interrupt enable bits off. So cancel any pending
00182     // interrupt and just return.
00183     deassertCSRInt();
00184     return;
00185   }
00186   // There is an interrupt enable on. Check for a source which has been
00187   // ready to trigger an interrupt.
00188   bool any_source_is_ready = false;
00189   for (int line = 0; line < 4; ++line)
00190     any_source_is_ready = any_source_is_ready
00191                           || (displayInterruptReadyForLine (line)
00192                               || keyboardInterruptReadyForLine (line));
00193   if (any_source_is_ready) {
00194     assertCSRInt();
00195   } else {
00196     deassertCSRInt();
00197   }
00198 }
