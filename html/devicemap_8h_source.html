00027 /* Physical memory map for a device supporting memory-mapped I/O. This is
00028  * an abstract class.
00029  */
00030 class DeviceMap : public Range {
00031 protected:
00032     DeviceMap() throw() : Range(0, 0, 0, 0) { }
00033     DeviceMap(uint32 _extent) throw() : Range(0, _extent, 0, 0) { }
00034 
00035 public:
00036     /* Fetch word at byte offset OFFSET as part of load type MODE, which
00037        can be either DATALOAD or INSTFETCH (defined in accesstypes.h).
00038        CLIENT is responsible for handling any exceptions the fetch
00039        may generate. */
00040     virtual uint32 fetch_word(uint32 offset, int mode, DeviceExc *client) = 0;
00041     
00042     /* Fetch halfword at byte offset OFFSET as part of a DATALOAD. CLIENT
00043        is responsible for handling any exceptions the fetch may generate.
00044        By default, calls fetch_word() and masks off the result. */
00045     virtual uint16 fetch_halfword(uint32 offset, DeviceExc *client);
00046 
00047     /* Fetch byte at byte offset OFFSET as part of a DATALOAD. CLIENT is
00048        responsible for handling any exceptions the fetch may generate. By
00049        default, calls fetch_word() and masks off the result. */
00050     virtual uint8 fetch_byte(uint32 offset, DeviceExc *client);
00051 
00052     /* Store word DATA at byte offset OFFSET as part of a DATASTORE.
00053        CLIENT is responsible for handling any exceptions the store may
00054        generate. */
00055     virtual void store_word(uint32 offset,uint32 data,DeviceExc *client) = 0;
00056 
00057     /* Store halfword DATA at byte offset OFFSET as part of a DATASTORE.
00058        Client is responsible for handling any exceptions the store may
00059        generate. By default, calls store_word() with a zeroed data word
00060        with DATA in the appropriate halfword. */
00061     virtual void store_halfword(uint32 offset, uint16 data, DeviceExc *client);
00062 
00063     /* Store byte DATA at byte offset OFFSET as part of a DATASTORE.
00064        Client is responsible for handing any exceptions the store may
00065        generate. By default, calls store_word() with a zeroed data word
00066        with DATA in the appropriate byte. */
00067     virtual void store_byte(uint32 offset, uint8 data, DeviceExc *client);
00068 
00069     /* It should not ordinarily be necessary for DeviceMap ranges to
00070        override these functions, except in the case where a device's mapped
00071        memory is properly conceived of as being either totally or
00072        partially read-only. See the definitions in class Range for details.  */
00073     virtual bool canRead(uint32 offset) throw();
00074     virtual bool canWrite(uint32 offset) throw();
00075 };
