00046 /* Add range R to the mapping. R must not overlap with any existing
00047  * ranges in the mapping. Return 0 if R added sucessfully or -1 if
00048  * R overlapped with an existing range. 
00049  */
00050 int
00051 Mapper::add_range(Range *r)
00052 {
00053         assert (r && "Null range object passed to Mapper::add_range()");
00054 
00055         /* Check to make sure the range is non-overlapping. */
00056         for (Ranges::iterator i = ranges.begin(); i != ranges.end(); i++) {
00057                 if (r->overlaps(*i)) {
00058                         error("Attempt to map two VMIPS components to the "
00059                                "same memory area: (base %x extent %x) and "
00060                                "(base %x extent %x).", r->getBase(), r->getExtent(),
00061                                (*i)->getBase(), (*i)->getExtent());
00062                         return -1;
00063                 }
00064         }
00065 
00066         /* Once we're satisfied that it doesn't overlap, add it to the list. */
00067         ranges.push_back(r);
00068         return 0;
00069 }

00071 /* Returns the first mapping in the rangelist, starting at the beginning,
00072  * which maps the address P, or NULL if no such mapping exists. This
00073  * function uses the LAST_USED_MAPPING instance variable as a cache to
00074  * speed a succession of accesses to the same area of memory.
00075  */
00076 Range *
00077 Mapper::find_mapping_range(uint32 p) throw()
00078 {
00079         if (last_used_mapping && last_used_mapping->incorporates(p))
00080                 return last_used_mapping;
00081 
00082         for (Ranges::iterator i = ranges.begin(), e = ranges.end(); i != e; ++i) {
00083                 if ((*i)->incorporates(p)) {
00084                         last_used_mapping = *i;
00085                         return *i;
00086                 }
00087         }
00088 
00089         return NULL;
00090 }

00101 /* Convert word W from big-endian to little-endian, or vice-versa,
00102  * and return the result of the conversion.
00103  */
00104 uint32
00105 Mapper::swap_word(uint32 w)
00106 {
00107         return ((w & 0x0ff) << 24) | (((w >> 8) & 0x0ff) << 16) |
00108         (((w >> 16) & 0x0ff) << 8) | ((w >> 24) & 0x0ff);
00109 }

00187 /* Fetch a word from the physical memory from physical address
00188  * ADDR. MODE is INSTFETCH if this is an instruction fetch; DATALOAD
00189  * otherwise. CACHEABLE is true if this access should be routed through
00190  * the cache, false otherwise.  This routine is shared between instruction
00191  * fetches and word-wide data fetches.
00192  * 
00193  * The routine returns either the specified word, if it is mapped and
00194  * the address is correctly aligned, or else a word consisting of all
00195  * ones is returned.
00196  *
00197  * Words are returned in the endianness of the target processor; since devices
00198  * are implemented as Ranges, devices should return words in the host
00199  * endianness.
00200  * 
00201  * This routine may trigger exceptions IBE and/or DBE in the client
00202  * processor, if the address is unmapped.
00203  * This routine may trigger exception AdEL in the client
00204  * processor, if the address is unaligned.
00205  */
00206 uint32
00207 Mapper::fetch_word(uint32 addr, int32 mode, bool cacheable, DeviceExc *client)
00208 {
00209         Range *l = NULL;
00210         uint32 offset;
00211 
00212         if (addr % 4 != 0) {
00213                 client->exception(AdEL,mode);
00214                 return 0xffffffff;
00215         }
00216         l = find_mapping_range(addr);
00217         if (!l) {
00218                 bus_error (client, mode, addr, 4);
00219                 return 0xffffffff;
00220         }
00221         offset = addr - l->getBase();
00222         if (!(l && l->canRead(offset))) {
00223                 /* Reads from nonexistent or write-only ranges return ones */
00224                 return 0xffffffff;
00225         }
00226         return host_to_mips_word(l->fetch_word(offset, mode, client));
00227 }

00410                 if (!dynamic_cast<MemoryModule *> (l)) {

00412                 } else {
00413                         for (int i = 0; i > -8; i--) {
00414                                 uint32 data =
00415                                         ((uint32 *) l->
00416                                          getAddress())[(stackphys - l->getBase()) / 4 + i];
00417                                 if (byteswapped)
00418                                         data = swap_word (data);
00419                                 fprintf(f, "%08x ", data);
00420                         }
00421                 }
00422         }
00423         fprintf(f, "\n");
