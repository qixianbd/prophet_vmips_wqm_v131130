00476                 {

00488                 if(head && m_ThreadVersion > Version(head))
00489                 {

02050                 faddr(swl),             faddr(sw),
02051                 faddr(spawn),           faddr(RI),
02052                 faddr(swr),             faddr(RI),

02055                 faddr(RI),              faddr(ldc1),
02056                 faddr(RI),              faddr(RI),
02057                 faddr(RI),              faddr(swc1),
02058                 faddr(swc2),            faddr(swc3),
02059                 faddr(RI),              faddr(sdc1),
02060                 faddr(RI),              faddr(RI),
02061         };
02062 
02063         pinfo<<(*this)<<"pc = "<<std::hex<<pc<<pendl;
02064         if(m_State != PRE_COMPUTATION && m_State != STABLE_EXECUTION && m_State != SP_EXECUTION)
02065         {
02066                 pinfo<<(*this)<<"not in a runnable state!"<<pendl;
02067                 return;
02068         }
02069 
02070         /* Clear exception_pending flag if it was set by a
02071          * prior instruction.
02072          */
02073         exception_pending = false;
02074 
02075         /* decrement Random register */
02076         cpzero->adjust_random();
02077 
02078         if(opt_tracing && !tracing && pc == opt_tracestartpc)
02079                 start_tracing();
02080 
02081         /* save address of instruction responsible for exceptions which may occur */
02082         if(delay_state != DELAYSLOT)
02083                 next_epc = pc;
02084 
02085         /* get physical address of next instruction */
02086         real_pc = cpzero->address_trans(pc, INSTFETCH, &cacheable, this);
02087         if (exception_pending) 
02088         {
02089                 if (opt_excmsg) 
02090                         pwarn<<(*this)<<"** PC address translation caused the exception! **"<<pendl;
02091                 goto out;
02092         }
02093 
02094         /* get next instruction */
02095         instr = mem->fetch_word(real_pc, INSTFETCH, cacheable, this);
02096         if(exception_pending)
02097         {
02098                 if(opt_excmsg) 
02099                         pwarn<<(*this)<<"** Instruction fetch caused the exception! **"<<pendl;
02100                 goto out;
02101         }
02102 
02103         /* diagnostic output - display disassembly of instr */
02104         if(opt_instdump)
02105         {
02106                 sprintf(out, "PC=0x%08x [%08x]\t%08x ", pc, real_pc, instr);
02107                 pinfo<<(*this)<<out<<pendl;
02108                 machine->disasm->disassemble(pc,instr);
02109         }
02110 
02111         /* first half of trace recording for this instruction */
02112         if(opt_tracing && tracing)
02113                 write_trace_record_1(pc, instr);
02114 
02115         /* Check for a (hardware or software) interrupt */
02116         if(cpzero->interrupt_pending())
02117         {
02118                 exception(Int);
02119                 goto out;
02120         }
02121 
02122 
02123         // 执行模拟函数*******************
02124         /* Jump to the appropriate emulation function. */
02125         sp_fcaller(OpcodeJumpTable[opcode(instr)], instr, pc);
02126 
02127 out:
02128         /* Register zero must always be zero; this instruction forces this. */
02129         WriteLocalReg(Sreg_zero,0);
02130 
02131         
02132         if(m_RegDumpMark)
02133         {
02134                 dump_regs(NULL);
02135         }
02136 
02137         /* 2nd half of trace recording for this instruction */
02138         if(opt_tracing && tracing)
02139         {
02140                 write_trace_record_2(pc, instr);
02141                 if (pc == opt_traceendpc)
02142                         stop_tracing();
02143         }
02144 
02145         /* If there is an exception pending, we return now, so that we don't
02146          * clobber the exception vector.
02147          */
02148         if(exception_pending)
02149         {
02150                 /* Instruction at beginning of exception handler is NOT in
02151                  * delay slot, no matter what the last instruction was.
02152                  */
02153                 delay_state = NORMAL;
02154                 return;
02155         }
02156 
02157         /* increment PC */
02158         if(delay_state == DELAYING)
02159         {
02160                 /* This instruction caused a branch to be taken.
02161                  * The next instruction is in the delay slot.
02162                  * The next instruction EPC will be PC - 4.
02163                  */
02164                 delay_state = DELAYSLOT;
02165                 pc = pc + 4;
02166         }else if(delay_state == DELAYSLOT){
02167                 /* This instruction was executed in a delay slot.
02168                  * The next instruction is on the other end of the branch.
02169                  * The next instruction EPC will be PC.
02170                  */
02171                 if(delay_pc == 0)
02172                 {
02173                         sprintf(out, "Jumped to zero (jump was at 0x%x)", pc - 4);
02174                         pwarn<<(*this)<<out<<pendl;
02175                 }
02176                 delay_state = NORMAL;
02177                 pc = delay_pc;
02178         }else if(delay_state == NORMAL){
02179                 /* No branch; next instruction is next word.
02180                  * Next instruction EPC is PC.
02181                  */
02182                 pc = pc + 4;
02183         }
02184 }
