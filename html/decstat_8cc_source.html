00035 #define ERRADR_VALID   0x80000000
00036 #define ERRADR_CPU     0x40000000
00037 #define ERRADR_WRITE   0x20000000
00038 #define ERRADR_ECCERR  0x10000000
00039 #define ERRADR_RSRVD   0x08000000
00040 #define ERRADR_ADDRESS 0x07FFFFFF

00042 uint32
00043 DECStatDevice::fetch_word (uint32 offset, int mode, DeviceExc *client)
00044 {
00045   uint32 rv;
00046   if (!(offset & 0x80000)) {
00047     fprintf (stderr, "CHKSYN reg read as 0x%x\n", chksyn_reg);
00048     rv = chksyn_reg;
00049   } else {
00050     Mapper::BusErrorInfo berr;
00051     machine->physmem->get_last_berr_info (berr);
00052     if (berr.valid) {
00053       erradr_reg = ERRADR_VALID;
00054       if (berr.mode == DATASTORE) erradr_reg |= ERRADR_WRITE;
00055       if (berr.client == dynamic_cast<DeviceExc *>(machine->cpu))
00056         erradr_reg |= ERRADR_CPU;
00057       // Simulate effects of pipelining - VMIPS's Mapper knows the
00058       // exact address, but the DECstation would have had to wait 5
00059       // cycles to get the address.
00060       berr.addr >>= 2;
00061       berr.addr = (berr.addr & ~0x0fffL) | ((berr.addr & 0x0fffL) + 5);
00062       berr.addr &= ERRADR_ADDRESS;
00063       erradr_reg |= berr.addr;
00064     } else {
00065       erradr_reg &= ~ERRADR_VALID;
00066     }
00067     fprintf (stderr, "ERRADR reg read as 0x%x\n", erradr_reg);
00068     rv = erradr_reg;
00069   }
00070   return rv;
00071 }

00073 void
00074 DECStatDevice::store_word (uint32 offset, uint32 data, DeviceExc *client)
00075 {
00076   if (!(offset & 0x80000)) {
00077     fprintf (stderr, "CHKSYN reg cleared\n");

00080     fprintf (stderr, "ERRADR reg cleared\n");
