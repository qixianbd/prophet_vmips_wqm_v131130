00048 void Clock::increment_time( long nanoseconds )
00049 {
00050         assert( nanoseconds > 0 );
00051         
00052         timespec increment = { 0, nanoseconds };
00053         timespecadd( &this->time, &increment );
00054 
00055         if( deferred_tasks.empty() )
00056                 return;
00057 
00058         for( DeferredTasks *tasks = deferred_tasks.front();
00059              !deferred_tasks.empty(); tasks = deferred_tasks.front() ) {
00060                 nanoseconds = tasks->pass_time( nanoseconds );
00061                 if( nanoseconds <= 0 ) {
00062                         deferred_tasks.pop_front();
00063                         delete tasks;
00064                 }
00065 
00066                 if( nanoseconds >= 0 )
00067                         break;
00068 
00069                 nanoseconds = -nanoseconds;
00070         }

00073 void Clock::pass_realtime( uint32 timeratio ) throw()
00074 {
00075         assert( timeratio > 0 );
00076         
00077         timeval current;
00078 
00079         // first time pass_realtime() has been called
00080         if( then.tv_sec == -1 ) {
00081                 gettimeofday( &current, NULL );
00082                 TIMEVAL_TO_TIMESPEC( &current, &then );
00083         }
00084 
00085         gettimeofday( &current, NULL );
00086         timespec now;
00087         TIMEVAL_TO_TIMESPEC( &current, &now );
00088 
00089         timespecsub( &now , &then );
00090         while( now.tv_sec > 0 ) {
00091                 increment_time( 1000000000 / timeratio );
00092                 now.tv_sec--;
00093         }
00094 
00095         long increase = (now.tv_nsec + spill_ns) / timeratio;
00096         if( increase == 0 )
00097                 spill_ns = now.tv_nsec + spill_ns;
00098         else {
00099                 spill_ns = 0;
00100                 increment_time( increase );
00101         }
00102 
00103         TIMEVAL_TO_TIMESPEC( &current, &then );
00104 }

00106 void Clock::add_deferred_task( Task *task, long nanoseconds )
00107 {
00108         assert( task );
00109         assert( nanoseconds > 0 );
00110         
00111         if( deferred_tasks.empty() ) {
00112                 DeferredTasks *t = new DeferredTasks( nanoseconds, task );
00113                 deferred_tasks.push_front( t );
00114                 return;
00115         }
00116 
00117         long wait_time = nanoseconds;
00118         for( list< DeferredTasks*>::iterator i = deferred_tasks.begin();
00119              i != deferred_tasks.end(); i++ ) {
00120                 wait_time = nanoseconds;
00121                 nanoseconds -= (*i)->get_nanoseconds_left();
00122                 
00123                 // insert a new set of deferred tasks before i
00124                 if( nanoseconds < 0 ) {
00125                         DeferredTasks *tasks=new DeferredTasks(wait_time,task);
00126                         deferred_tasks.insert( i, tasks );
00127 
00128                         // decrement the waiting time for the next task by
00129                         // wait_time so the task executes at the right time
00130                         (*i)->pass_time( wait_time );
00131 
00132                         return;
00133                 }
00134                 // the wait time for TASK is the same as these tasks
00135                 else if( nanoseconds == 0 ) {
00136                         (*i)->add_task( task );
00137                         return;
00138                 }
00139         }
00140         
00141         DeferredTasks *tasks = new DeferredTasks( nanoseconds, task );
00142         deferred_tasks.push_back( tasks );
00143 }
