00026 /* This file consists almost entirely of code taken from the GNU debugger
00027    (gdb) version 4.17, from file gdb/remote.c.  Additions by me (brg) have
00028    been tagged with my initials.

00030    See the GDB User Guide for the definitive details of the GDB remote
00031    protocol.  The following notes correspond to the protocol for GDB
00032    version 4.17.
00033 
00034    Remote communication protocol.
00035 
00036    A debug packet whose contents are <data>
00037    is encapsulated for transmission in the form:
00038 
00039         $ <data> # CSUM1 CSUM2
00040 
00041         <data> must be ASCII alphanumeric and cannot include characters
00042         '$' or '#'.  If <data> starts with two characters followed by
00043         ':', then the existing stubs interpret this as a sequence number.
00044 
00045         CSUM1 and CSUM2 are ascii hex representation of an 8-bit 
00046         checksum of <data>, the most significant nibble is sent first.
00047         the hex digits 0-9,a-f are used.
00048 
00049    Receiver responds with:
00050 
00051         +       - if CSUM is correct and ready for next packet
00052         -       - if CSUM is incorrect
00053 
00054    <data> is as follows:
00055    Most values are encoded in ascii hex digits.  Signal numbers are according
00056    to the numbering in target.h.
00057 
00058         Request         Packet
00059 
00060         set thread      Hct...          Set thread for subsequent operations.
00061                                         c = 'c' for thread used in step and 
00062                                         continue; t... can be -1 for all
00063                                         threads.
00064                                         c = 'g' for thread used in other
00065                                         operations.  If zero, pick a thread,
00066                                         any thread.
00067         reply           OK              for success
00068                         ENN             for an error.
00069 
00070         read registers  g
00071         reply           XX....X         Each byte of register data
00072                                         is described by two hex digits.
00073                                         Registers are in the internal order
00074                                         for GDB, and the bytes in a register
00075                                         are in the same order the machine uses.
00076                         or ENN          for an error.
00077 
00078         write regs      GXX..XX         Each byte of register data
00079                                         is described by two hex digits.
00080         reply           OK              for success
00081                         ENN             for an error
00082 
00083         write reg       Pn...=r...      Write register n... with value r...,
00084                                         which contains two hex digits for each
00085                                         byte in the register (target byte
00086                                         order).
00087         reply           OK              for success
00088                         ENN             for an error
00089         (not supported by all stubs).
00090 
00091         read mem        mAA..AA,LLLL    AA..AA is address, LLLL is length.
00092         reply           XX..XX          XX..XX is mem contents
00093                                         Can be fewer bytes than requested
00094                                         if able to read only part of the data.
00095                         or ENN          NN is errno
00096 
00097         write mem       MAA..AA,LLLL:XX..XX
00098                                         AA..AA is address,
00099                                         LLLL is number of bytes,
00100                                         XX..XX is data
00101         reply           OK              for success
00102                         ENN             for an error (this includes the case
00103                                         where only part of the data was
00104                                         written).
00105 
00106         continue        cAA..AA         AA..AA is address to resume
00107                                         If AA..AA is omitted,
00108                                         resume at same address.
00109 
00110         step            sAA..AA         AA..AA is address to resume
00111                                         If AA..AA is omitted,
00112                                         resume at same address.
00113 
00114         continue with   Csig;AA..AA     Continue with signal sig (hex signal
00115         signal                          number).  If ;AA..AA is omitted, resume
00116                                         at same address.
00117 
00118         step with       Ssig;AA..AA     Like 'C' but step not continue.
00119         signal
00120 
00121         last signal     ?               Reply the current reason for stopping.
00122                                         This is the same reply as is generated
00123                                         for step or cont : SAA where AA is the
00124                                         signal number.
00125 
00126         detach          D               Reply OK.
00127 
00128         There is no immediate reply to step or cont.
00129         The reply comes when the machine stops.
00130         It is           SAA             AA is the signal number.
00131 
00132         or...           TAAn...:r...;n...:r...;n...:r...;
00133                                         AA = signal number
00134                                         n... = register number (hex)
00135                                           r... = register contents
00136                                         n... = `thread'
00137                                           r... = thread process ID.  This is
00138                                                  a hex integer.
00139                                         n... = other string not starting 
00140                                             with valid hex digit.
00141                                           gdb should ignore this n,r pair
00142                                           and go on to the next.  This way
00143                                           we can extend the protocol.
00144         or...           WAA             The process exited, and AA is
00145                                         the exit status.  This is only
00146                                         applicable for certains sorts of
00147                                         targets.
00148         or...           XAA             The process terminated with signal
00149                                         AA.
00150         or...           OXX..XX XX..XX  is hex encoding of ASCII data. This
00151                                         can happen at any time while the program is
00152                                         running and the debugger should
00153                                         continue to wait for 'W', 'T', etc.
00154 
00155         thread alive    TXX             Find out if the thread XX is alive.
00156         reply           OK              thread is still alive
00157                         ENN             thread is dead
00158         
00159         remote restart  RXX             Restart the remote server
00160 
00161         extended ops    !               Use the extended remote protocol.
00162                                         Sticky -- only needs to be set once.
00163 
00164         kill request    k
00165 
00166         toggle debug    d               toggle debug flag (see 386 & 68k stubs)
00167         reset           r               reset -- see sparc stub.
00168         reserved        <other>         On other requests, the stub should
00169                                         ignore the request and send an empty
00170                                         response ($#<checksum>).  This way
00171                                         we can extend the protocol and GDB
00172                                         can tell whether the stub it is
00173                                         talking to uses the old or the new.
00174         search          tAA:PP,MM       Search backwards starting at address
00175                                         AA for a match with pattern PP and
00176                                         mask MM.  PP and MM are 4 bytes.
00177                                         Not supported by all stubs.
00178 
00179         general query   qXXXX           Request info about XXXX.
00180         general set     QXXXX=yyyy      Set value of XXXX to yyyy.
00181         query sect offs qOffsets        Get section offsets.  Reply is
00182                                         Text=xxx;Data=yyy;Bss=zzz
00183 
00184         Responses can be run-length encoded to save space.  A '*' means that
00185         the next character is an ASCII encoding giving a repeat count which
00186         stands for that many repititions of the character preceding the '*'.
00187         The encoding is n+29, yielding a printable character where n >=3 
00188         (which is where rle starts to win).  Don't use an n > 126.
00189 
00190         So 
00191         "0* " means the same as "0000".  */

00603 /* Read a packet from the remote machine, with error checking,
00604    and store it in BUF.  BUF is expected to be of size PBUFSIZ.
00605    If FOREVER, wait forever rather than timing out; this is used
00606    while the target is executing user code.  */

00608 void getpkt (char *buf, int forever)
00609 {
00610   int c;
00611   int tries;
00612   int timeout;
00613   int val;
00614 
00615   strcpy (buf,"timeout");
00616 
00617   if (forever)
00618     {
00619 #ifdef MAINTENANCE_CMDS
00620       timeout = watchdog > 0 ? watchdog : -1;
00621 #else
00622       timeout = -1;
00623 #endif
00624     }
00625 
00626   else
00627     timeout = remote_timeout;
00628 
00629 #define MAX_TRIES 3
00630 
00631   for (tries = 1; tries <= MAX_TRIES; tries++)
00632     {
00633       /* This can loop forever if the remote side sends us characters
00634          continuously, but if it pauses, we'll get a zero from readchar
00635          because of timeout.  Then we'll count that as a retry.  */
00636 
00637       /* Note that we will only wait forever prior to the start of a packet.
00638          After that, we expect characters to arrive at a brisk pace.  They
00639          should show up within remote_timeout intervals.  */
00640 
00641       do
00642         {
00643           c = readchar (timeout);
00644 
00645           if (c == SERIAL_TIMEOUT)
00646             {
00647 #ifdef MAINTENANCE_CMDS
00648               if (forever)      /* Watchdog went off.  Kill the target. */
00649                 {
00650                   target_mourn_inferior ();
00651                   error ("Watchdog has expired.  Target detached.\n");
00652                 }
00653 #endif
00654               if (remote_debug)
00655                 puts_filtered ("Timed out.\n");
00656               goto retry;
00657             }
00658         }
00659       while (c != '$');
00660 
00661       /* We've found the start of a packet, now collect the data.  */
00662 
00663       val = read_frame (buf);
00664 
00665       if (val == 1)
00666         {
00667           if (remote_debug)
00668             fprintf_unfiltered (gdb_stdout, "Packet received: %s\n", buf);
00669           SERIAL_WRITE (remote_desc, "+", 1);
00670           return;
00671         }
00672 
00673       /* Try the whole thing again.  */
00674     retry:
00675       SERIAL_WRITE (remote_desc, "-", 1);
00676     }
00677 
00678   /* We have tried hard enough, and just can't receive the packet.  Give up. */
00679 
00680   printf_unfiltered ("Ignoring packet error, continuing...\n");
00681   SERIAL_WRITE (remote_desc, "+", 1);
00682 }
