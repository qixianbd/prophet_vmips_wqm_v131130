00133 /* Connect the file or device named NAME to line number L of
00134  * console device C, or do nothing if NAME is "off".
00135  */
00136 void vmips::setup_console_line(int l, char *name, TerminalController *c, const
00137 char *c_name) throw()
00138 {
00139         /* If they said to turn off the tty line, do nothing. */
00140         if (strcmp(name, "off") == 0)
00141                 return;
00142 
00143         /* Open the file or device in question. */
00144         int ttyfd = open(name, O_RDWR | O_NONBLOCK);
00145         if (ttyfd == -1) {
00146                 /* If we can't open it, warn and use stdout instead. */
00147                 error("Opening %s (terminal %d): %s", name, l, strerror(errno));
00148                 warning("using stdout, input disabled\n");
00149                 ttyfd = fileno(stdout);
00150         }
00151 
00152         /* Connect it to the SPIM-compatible console device. */
00153         c->connect_terminal(ttyfd, l);
00154         boot_msg("Connected fd %d to %s line %d.\n", ttyfd, c_name, l);
00155 }

00348 vmips::step(void)
00349 {
00350         /* Process instructions. */
00351         cpu->step();
00352 
00353         /* Keep track of time passing. Each instruction either takes
00354          * clock_nanos nanoseconds, or we use pass_realtime() to check the
00355          * system clock.
00356      */
00357         if( !opt_realtime )
00358            clock->increment_time(clock_nanos);
00359         else
00360            clock->pass_realtime(opt_timeratio);
00361 
00362         /* If user requested it, dump registers from CPU and/or CP0. */
00363         if (opt_dumpcpu)
00364                 cpu->dump_regs_and_stack(stderr);
00365         if (opt_dumpcp0)
00366                 cpu->cpzero_dump_regs_and_tlb(stderr);
00367         num_instrs++;
