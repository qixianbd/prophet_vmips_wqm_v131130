00026 /* Interrupt lines that DeviceInts can use.
00027    These constants correspond to the bits of the Interrupt Pending (IP)
00028    field of the Cause register of CP0, and to the bits of the Interrupt Mask
00029    (IM) field of the Status register of CP0. See cpzeroreg.h.  */
00030 #define IRQ7 0x00008000
00031 #define IRQ6 0x00004000
00032 #define IRQ5 0x00002000
00033 #define IRQ4 0x00001000
00034 #define IRQ3 0x00000800
00035 #define IRQ2 0x00000400
00036 
00037 /* These are for use by software only, NOT hardware! */
00038 #define IRQ1 0x00000200
00039 #define IRQ0 0x00000100

00041 /* A class which describes a device that can trigger hardware interrupts
00042    in the processor. DeviceInts can be connected to an IntCtrl
00043    (interrupt controller). */
00044 class DeviceInt {
00045 public:
00046         /* Given a value (LINE) representing one of the bits of the
00047            Cause register (IRQ7 .. IRQ0 in deviceint.h), return a string
00048            that describes the corresponding interrupt line.  The string is
00049            returned in a static buffer, so the next call to strlineno will
00050            overwrite the result. */
00051         static char *strlineno(uint32 line);
00052 
00053         /* Return the bit of the CP0 Status and Cause registers corresponding
00054            to interrupt line number NUM, or 0 if no such interrupt line exists. */
00055         static uint32 num2irq(uint32 num) throw();
00056 
00057     virtual ~DeviceInt() { }
00058 
00059 protected:
00060         /* Assert an interrupt request on interrupt line LINE, which
00061            must be one of the IRQ7 .. IRQ0 constants in deviceint.h. An
00062            interrupt thus asserted remains asserted until it is explicitly
00063            deasserted. */
00064         void assertInt(uint32 line);
00065 
00066         /* Deassert an interrupt request on interrupt line LINE, which
00067            must be one of the IRQ7 .. IRQ0 constants in deviceint.h. Note
00068            that if one device deasserts the interrupt request, that doesn't
00069            necessarily mean that the interrupt line will go to zero;
00070            another device may be sharing the same interrupt line. */
00071         void deassertInt(uint32 line);
00072 
00073         /* This method, in derived classes, should return a string describing
00074            the device. */
00075         virtual const char *descriptor_str(void) const = 0;
00076 
00077         DeviceInt() throw();
00078 
00079 private:
00080         void reportAssert(uint32 line);
00081         void reportAssertDisconnected(uint32 line);
00082         void reportDeassert(uint32 line);
00083         void reportDeassertDisconnected(uint32 line);
00084 
00085         friend class IntCtrl;
00086 
00087         uint32 lines_connected;
00088         uint32 lines_asserted;
00089         bool opt_reportirq;
00090 };
