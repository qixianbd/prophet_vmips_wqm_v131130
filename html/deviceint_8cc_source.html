00027 /* Given a value (LINE) representing one of the bits of the
00028    Cause register (IRQ7 .. IRQ0 in deviceint.h), return
00029    a string that describes the corresponding interrupt line.
00030    The string is returned in a static buffer, so the next call
00031    to strlineno will overwrite the result. */
00032 char *DeviceInt::strlineno(uint32 line)
00033 {
00034         static char buff[50];
00035 
00036         if (line == IRQ7) { sprintf(buff, "IRQ7"); }
00037         else if (line == IRQ6) { sprintf(buff, "IRQ6"); }
00038         else if (line == IRQ5) { sprintf(buff, "IRQ5"); }
00039         else if (line == IRQ4) { sprintf(buff, "IRQ4"); }
00040         else if (line == IRQ3) { sprintf(buff, "IRQ3"); }
00041         else if (line == IRQ2) { sprintf(buff, "IRQ2"); }
00042         else if (line == IRQ1) { sprintf(buff, "IRQ1"); }
00043         else if (line == IRQ0) { sprintf(buff, "IRQ0"); }
00044         else { sprintf(buff, "something strange (0x%08x)", line); }
00045         return buff;
00046 }

00107 /* Assert an interrupt request on interrupt line LINE, which must be one
00108    of the IRQ7 .. IRQ0 constants in deviceint.h. An interrupt thus asserted
00109    remains asserted until it is explicitly deasserted. */
00110 void DeviceInt::assertInt(uint32 line)
00111 {
00112         if (line & lines_connected) {
00113                 if (! (line & lines_asserted)) reportAssert(line);
00114                 lines_asserted |= line;
00115         } else {
00116                 reportAssertDisconnected(line);
00117         }
00118 }

00120 /* Deassert an interrupt request on interrupt line LINE, which must be one
00121    of the IRQ7 .. IRQ0 constants in deviceint.h. Note that if one device
00122    deasserts the interrupt request, that doesn't necessarily mean that
00123    the interrupt line will go to zero; another device may be sharing the
00124    same interrupt line. */
00125 void DeviceInt::deassertInt(uint32 line)
00126 {
00127         if (line & lines_connected) {
00128                 if (line & lines_asserted) reportDeassert(line);
00129                 lines_asserted &= ~line;
00130         } else {
00131                 reportDeassertDisconnected(line);
00132         }
00133 }
