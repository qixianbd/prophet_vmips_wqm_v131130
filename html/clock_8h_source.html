00029 /* The Clock class manages virtual time. */
00030 class Clock
00031 {
00032 public:
00033         /* Create a new simulation clock with starting time START_TIME, which
00034            should be a positive starting time. */
00035         Clock( const timespec &start_time ) throw();
00036 
00037         /* Destroy the clock object and delete all queued tasks. */
00038         virtual ~Clock() throw();

00040         /* Move simulated time forward NANOSECONDS nanoseconds and execute
00041            any deferred tasks that have their expiration date before the
00042            new time. */
00043         virtual void increment_time( long nanoseconds );
00044 
00045         /* Increment the simulated clock by the difference from the last
00046            call to pass_realtime() and the current time (as determined by
00047            gettimeofday(2)) divided by timeratio. */
00048         virtual void pass_realtime( uint32 timeratio ) throw();
00049 
00050         /* Queue task TASK to be executed in NANOSECONDS nanoseconds from the
00051            present time. TASK's task() function will be called and then it
00052            will be delete'd (possibly after other task's task() functions
00053            have been called). */
00054         virtual void add_deferred_task( Task *task, long nanoseconds );
00055 
00056         /* Return the simulated time as a timespec. */
00057         virtual timespec get_time() throw();
00058 
00059         /* Set the simulated time to TIME. All components of TIME should be
00060            non-negative. */
00061         virtual void set_time( const timespec &time ) throw();
00062 
00063 protected:
00064         /* Each DeferredTasks object is responsible for maintaining a list of
00065            tasks to execute in the future. */
00066         class DeferredTasks
00067         {
00068         public:
00069                 /* Create a new DeferredTasks that will come due in
00070                    NANOSECONDS_LEFT nanoseconds from when it starts
00071                    pass_time(). Will execute task TASK and any other tasks
00072                    added later via add_task() in a fifo manner when it comes
00073                    due. */
00074                 DeferredTasks( long nanoseconds_left, Task *task ) throw();
00075 
00076                 /* Destory deferred tasks by delete'ing each managed task. */
00077                 virtual ~DeferredTasks() throw();
00078 
00079                 /* Add task TASK to the list of tasks to complete when the
00080                    deferred tasks come due. TASK will be delete'd after it is
00081                    run. */
00082                 virtual void add_task( Task *task ) throw();
00083         
00084                 /* Mark the passage of NANOSECONDS nanoseconds. If the
00085                    deferred tasks come due then execute the tasks in the order
00086                    in which they where added. Returns a number of nanoseconds
00087                    that, if positive, is the number of nanoseconds left until
00088                    the deferred tasks come due and if negative, is the number
00089                    of nanoseconds remaining after the deferred tasks object
00090                    has used NANOSECONDS. */
00091                 virtual long pass_time( long nanoseconds );
00092 
00093                 /* Return the number of nanoseconds of simulated time left

00095                 inline virtual long get_nanoseconds_left() throw();
00096 
00097         protected:
00098                 long                    nanoseconds_left;
00099                 std::list< Task* >      tasks;
00100         };
