00237         struct excPriority *p, prio[] = {
00238                 {1, AdEL, INSTFETCH},
00239                 {2, TLBL, INSTFETCH}, {2, TLBS, INSTFETCH},
00240                 {3, IBE, ANY},
00241                 {4, Ov, ANY}, {4, Tr, ANY}, {4, Sys, ANY},
00242                 {4, Bp, ANY}, {4, RI, ANY}, {4, CpU, ANY},

00247                 {8, DBE, ANY},

00249                 {0, ANY, ANY} /* catch-all */

00269 void
00270 CPU::exception(uint16 excCode, int mode /* = ANY */, int coprocno /* = -1 */)
00271 {
00272         int prio;
00273         uint32 base, vector, epc;
00274         bool delaying = (delay_state == DELAYSLOT);
00275 
00276         if (opt_haltbreak) {
00277                 if (excCode == Bp) {
00278                         fprintf(stderr,"* BREAK instruction reached -- HALTING *\n");
00279                         machine->halt();
00280                 }
00281         }
00282         if (opt_haltibe) {
00283                 if (excCode == IBE) {
00284                         fprintf(stderr,"* Instruction bus error occurred -- HALTING *\n");
00285                         machine->halt();
00286                 }
00287         }
00288 
00289         /* step() ensures that next_epc will always contain the correct
00290          * EPC whenever exception() is called.
00291          */
00292         epc = next_epc;
00293 
00294         /* Prioritize exception -- if the last exception to occur _also_ was
00295          * caused by this EPC, only report this exception if it has a higher
00296          * priority.  Otherwise, exception handling terminates here,
00297          * because only one exception will be reported per instruction
00298          * (as per MIPS RISC Architecture, p. 6-35). Note that this only
00299          * applies IFF the previous exception was caught during the current
00300          * _execution_ of the instruction at this EPC, so we check that
00301          * EXCEPTION_PENDING is true before aborting exception handling.
00302          * (This flag is reset by each call to step().)
00303          */
00304         prio = exception_priority(excCode, mode);
00305         if (epc == last_epc) {
00306                 if (prio <= last_prio && exception_pending) {
00307                         if (opt_excpriomsg) {
00308                                 fprintf(stderr,
00309                                         "(Ignoring additional lower priority exception...)\n");
00310                         }
00311                         return;
00312                 } else {
00313                         last_prio = prio;
00314                 }
00315         }
00316         last_epc = epc;
00317 
00318         /* Set processor to Kernel mode, disable interrupts, and save 
00319          * exception PC.
00320          */
00321         cpzero->enter_exception(epc,excCode,coprocno,delaying);
00322 
00323         /* Calculate the exception handler address; this is of the form BASE +
00324          * VECTOR. The BASE is determined by whether we're using boot-time
00325          * exception vectors, according to the BEV bit in the CP0 Status register.
00326          */
00327         if (cpzero->use_boot_excp_address()) {
00328                 base = 0xbfc00100;
00329         } else {
00330                 base = 0x80000000;
00331         }
00332         
00333 
00334         /* Do we have a User TLB Miss exception? If so, jump to the
00335          * User TLB Miss exception vector, otherwise jump to the
00336          * common exception vector.
00337          */
00338         if ((excCode == TLBL || excCode == TLBS) && (cpzero->tlb_miss_user)) {
00339                 vector = 0x000;
00340         } else {
00341                 vector = 0x080;
00342         }
00343 
00344         if (opt_excmsg && (excCode != Int || opt_reportirq)) {
00345                 fprintf(stderr,"Exception %d (%s) triggered, EPC=%08x\n", excCode, 
00346                         strexccode(excCode), epc);
00347                 fprintf(stderr,
00348                         " Priority is %d; delay state is %s; mem access mode is %s\n",
00349                         prio, strdelaystate(delay_state), strmemmode(mode));
00350                 if (excCode == Int) {
00351                   uint32 status, bad, cause;
00352                   cpzero->read_debug_info(&status, &bad, &cause);
00353                   fprintf (stderr, " Interrupt cause = %x, status = %x\n", cause, status);
00354                 }
00355         }
00356     if (opt_tracing) {
00357                 if (tracing) {
00358                         current_trace.exception_happened = true;
00359                         current_trace.last_exception_code = excCode;
00360                 }
00361         }
00362         pc = base + vector;
00363         exception_pending = true;
00364 }

00635 /* The lwr and lwl algorithms here are taken from SPIM 6.0,
00636  * since I didn't manage to come up with a better way to write them.
00637  * Improvements are welcome.
00638  */
00639 uint32
00640 CPU::lwr(uint32 regval, uint32 memval, uint8 offset)
00641 {
00642         if (opt_bigendian) {
00643                 switch (offset)
00644                 {
00645                         case 0: return (regval & 0xffffff00) |
00646                                                 ((unsigned)(memval & 0xff000000) >> 24);
00647                         case 1: return (regval & 0xffff0000) |
00648                                                 ((unsigned)(memval & 0xffff0000) >> 16);
00649                         case 2: return (regval & 0xff000000) |
00650                                                 ((unsigned)(memval & 0xffffff00) >> 8);
00651                         case 3: return memval;
00652                 }
00653         } else /* if MIPS target is little endian */ {
00654                 switch (offset)
00655                 {
00656                         /* The SPIM source claims that "The description of the
00657                          * little-endian case in Kane is totally wrong." The fact
00658                          * that I ripped off the LWR algorithm from them could be
00659                          * viewed as a sort of passive assumption that their claim
00660                          * is correct.
00661                          */
00662                         case 0: /* 3 in book */
00663                                 return memval;
00664                         case 1: /* 0 in book */
00665                                 return (regval & 0xff000000) | ((memval & 0xffffff00) >> 8);
00666                         case 2: /* 1 in book */

01780                 &CPU::RI_emulate,    &CPU::RI_emulate,

01782         };
01783 
01784         /* Clear exception_pending flag if it was set by a
01785          * prior instruction.
01786          */
01787         exception_pending = false;
01788 
01789         /* decrement Random register */
01790         cpzero->adjust_random();
01791 
01792         if (opt_tracing && !tracing && pc == opt_tracestartpc)
01793                 start_tracing();
01794 
01795         /* save address of instruction responsible for exceptions which may occur */
01796         if (delay_state != DELAYSLOT) {
01797                 next_epc = pc;
01798         }
01799 
01800         /* get physical address of next instruction */
01801         real_pc = cpzero->address_trans(pc,INSTFETCH,&cacheable, this);
01802         if (exception_pending) {
01803                 if (opt_excmsg) {
01804                         fprintf(stderr,
01805                                 "** PC address translation caused the exception! **\n");
01806                 }
01807                 goto out;
01808         }
01809 
01810         /* get next instruction */
01811         instr = mem->fetch_word(real_pc,INSTFETCH,cacheable, this);
01812         if (exception_pending) {
01813                 if (opt_excmsg) {
01814                         fprintf(stderr, "** Instruction fetch caused the exception! **\n");
01815                 }
01816                 goto out;
01817         }
01818 
01819         /* diagnostic output - display disassembly of instr */
01820         if (opt_instdump) {
01821                 fprintf(stderr,"PC=0x%08x [%08x]\t%08x ",pc,real_pc,instr);
01822                 machine->disasm->disassemble(pc,instr);
01823         }
01824 
01825         /* first half of trace recording for this instruction */
01826         if (opt_tracing && tracing)
01827                 write_trace_record_1 (pc, instr);
01828 
01829         /* Check for a (hardware or software) interrupt */
01830         if (cpzero->interrupt_pending()) {
01831                 exception(Int);
01832                 goto out;
01833         }
01834 
01835         /* Jump to the appropriate emulation function. */
01836         (this->*opcodeJumpTable[opcode(instr)])(instr, pc);
01837 
01838 out:
01839         /* Register zero must always be zero; this instruction forces this. */
01840         reg[reg_zero] = 0;
01841 
01842         /* 2nd half of trace recording for this instruction */
01843         if (opt_tracing && tracing) {
01844                 write_trace_record_2 (pc, instr);
01845                 if (pc == opt_traceendpc)
01846                         stop_tracing();
01847         }
01848 
01849         /* If there is an exception pending, we return now, so that we don't
01850          * clobber the exception vector.
01851          */
01852         if (exception_pending) {
01853                 /* Instruction at beginning of exception handler is NOT in
01854                  * delay slot, no matter what the last instruction was.
01855                  */
01856                 delay_state = NORMAL;
01857                 return;
01858         }
01859 
01860         /* increment PC */
01861         if (delay_state == DELAYING) {
01862                 /* This instruction caused a branch to be taken.
01863                  * The next instruction is in the delay slot.
01864                  * The next instruction EPC will be PC - 4.
01865                  */
01866                 delay_state = DELAYSLOT;
01867                 pc = pc + 4;
01868         } else if (delay_state == DELAYSLOT) {
01869                 /* This instruction was executed in a delay slot.
01870                  * The next instruction is on the other end of the branch.
01871                  * The next instruction EPC will be PC.
01872                  */
01873                 if (delay_pc == 0)
01874                         warning ("Jumped to zero (jump was at 0x%x)\n", pc - 4);
01875                 delay_state = NORMAL;
01876                 pc = delay_pc;
01877         } else if (delay_state == NORMAL) {
01878                 /* No branch; next instruction is next word.
01879                  * Next instruction EPC is PC.
01880                  */
01881                 pc = pc + 4;
01882         }
01883 }
